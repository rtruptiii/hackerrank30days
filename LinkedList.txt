import java.io.*;
import java.util.*;

class Node {
	int data;
	Node next;
	Node(int d) {
        data = d;
        next = null;
    }
}

class Solution {

    public static  Node insert(Node head,int data) {
        //Complete this method
        
    if(head == null)
        return new Node( data);
    
    if(head.next == null){
            head.next = new Node(data);
        }
    else{
        insert(head.next,data);
    }

    return head;
}
